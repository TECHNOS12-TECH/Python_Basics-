#Assignment operators
a=float(input("ENTER THE VALUE OF a:"))
b=float(input("Enter THE VALUE OF b:"))
c=float(input("ENTER THE VALUE OF c:"))
d=float(input("ENTER THE VALUE OF d:"))
a+=10
b-=10
c*=10
d/=10
print("THE VALUE OF a WITH ADDITION OF 10 is:",a)
print("THE VALUE OF b WITH SUBTRACTION OF 10 is:",b)
print("THE VALUE OF c WITH MULTIPLICATION OF 10 is:",c)
print("THE VALUE OF d WITH DIVISION OF 10 is:",d)
